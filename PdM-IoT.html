def add_time_features(sensor_df, asset_metadata):
    sensor_merged = pd.merge(sensor_df, asset_metadata[['machine_id', 'install_date']], on='machine_id', how='left')
    sensor_merged['days_since_installation'] = (sensor_merged['timestamp'] - sensor_merged['install_date']).dt.total_seconds() / (24 * 3600)
    
    # Add hour of day and day of week
    sensor_merged['hour_of_day'] = sensor_merged['timestamp'].dt.hour
    sensor_merged['day_of_week'] = sensor_merged['timestamp'].dt.dayofweek
    
    # Convert categorical variables to one-hot encoding
    sensor_merged = pd.get_dummies(sensor_merged, columns=['shift'], drop_first=True)
    
    return sensor_merged

def label_failures(sensor_merged, failure_df, early_window, imminent_window):
    for _, row in tqdm(failure_df.iterrows(), total=len(failure_df), desc="Labeling failures"):
        machine_id = row["machine_id"]
        failure_time = row["failure_time"]
        
        # Imminent failure
        imminent_mask = (
            (sensor_merged["machine_id"] == machine_id) &
            (sensor_merged["timestamp"] >= (failure_time - imminent_window)) &
            (sensor_merged["timestamp"] < failure_time)
        )
        sensor_merged.loc[imminent_mask, "failure_warning"] = 2
        
        # Early warning
        early_mask = (
            (sensor_merged["machine_id"] == machine_id) &
            (sensor_merged["timestamp"] >= (failure_time - early_window)) &
            (sensor_merged["timestamp"] < (failure_time - imminent_window))
        )
        sensor_merged.loc[early_mask, "failure_warning"] = 1
    
    return sensor_merged

# Advanced feature engineering with multiple time windows
def extract_features(sensor_merged):
    non_features = ['timestamp', 'machine_id', 'failure_warning', 'install_date']
    numeric_cols = sensor_merged.drop(columns=non_features, errors='ignore').select_dtypes(include=np.number).columns.tolist()
    
    # Windows for different timeframes
    windows = ['5min', '15min', '30min', '1H', '3H']
    
    all_machine_features = []
    
    for machine_id, df_group in tqdm(sensor_merged.groupby("machine_id"), desc="Extracting features"):
        df_group = df_group.set_index("timestamp").sort_index()
        
        machine_features = []
        
        for window in windows:
            # Rolling window stats
            rolled = df_group[numeric_cols].rolling(window=window)
            
            # Basic statistics
            stats = rolled.agg(['mean', 'std', 'min', 'max']).dropna()
            
            # Advanced features
            # Rate of change for key indicators
            for col in ['rpm', 'vibration', 'motor_temp', 'current', 'oil_temp']:
                if col in df_group.columns:
                    # Compute rate of change (derivative)
                    stats[f'{col}_rate_of_change'] = df_group[col].diff().rolling(window=window).mean().loc[stats.index]
                    
                    # Compute acceleration (second derivative)
                    stats[f'{col}_acceleration'] = df_group[col].diff().diff().rolling(window=window).mean().loc[stats.index]
                    
                    # Compute relative change
                    stats[f'{col}_rel_change'] = (df_group[col].pct_change().rolling(window=window).mean() * 100).loc[stats.index]
            
            # Flatten multi-index columns
            stats.columns = [f'{col[0]}_{col[1]}_{window}' if isinstance(col, tuple) else f'{col}_{window}' for col in stats.columns]
            
            machine_features.append(stats)
        
        # Combine features from all windows
        if machine_features:
            combined = pd.concat(machine_features, axis=1)
            
            # Keep the original machine_id and labels
            combined["machine_id"] = machine_id
            combined["timestamp"] = combined.index
            combined["failure_warning"] = df_group.loc[combined.index, "failure_warning"].values
            
            all_machine_features.append(combined.reset_index(drop=True))
    
    # Combine all into a single DataFrame
    if all_machine_features:
        final_df = pd.concat(all_machine_features).reset_index(drop=True)
        return final_df
    else:
        return pd.DataFrame()

# Add failure history per machine
def add_failure_history(feature_df, failure_df):
    # Sort failure logs by time
    failure_df = failure_df.sort_values('failure_time')
    
    # Create a dict to track recent failures per machine
    recent_failures = {machine: 0 for machine in feature_df['machine_id'].unique()}
    failure_history = []
    
    # For each row in feature_df
    for _, row in tqdm(feature_df.iterrows(), total=len(feature_df), desc="Adding failure history"):
        machine_id = row['machine_id']
        current_time = row['timestamp']
        
        # Count failures in last 7 days
        failures_7d = failure_df[
            (failure_df['machine_id'] == machine_id) & 
            (failure_df['failure_time'] < current_time) & 
            (failure_df['failure_time'] >= current_time - pd.Timedelta(days=7))
        ].shape[0]
        
        # Count failures in last 30 days
        failures_30d = failure_df[
            (failure_df['machine_id'] == machine_id) & 
            (failure_df['failure_time'] < current_time) & 
            (failure_df['failure_time'] >= current_time - pd.Timedelta(days=30))
        ].shape[0]
        
        # Time since last failure
        last_failure = failure_df[
            (failure_df['machine_id'] == machine_id) & 
            (failure_df['failure_time'] < current_time)
        ]
        
        if not last_failure.empty:
            days_since_last_failure = (current_time - last_failure['failure_time'].max()).days
        else:
            days_since_last_failure = 365  # Default value if no previous failure
        
        failure_history.append({
            'failures_7d': failures_7d,
            'failures_30d': failures_30d,
            'days_since_last_failure': days_since_last_failure
        })
    
    failure_history_df = pd.DataFrame(failure_history)
    feature_df = pd.concat([feature_df, failure_history_df], axis=1)
    
    return feature_df

# Time-based split with temporal buffer
def temporal_split(df, train_ratio=0.7, buffer_ratio=0.1):
    # Sort by timestamp
    df = df.sort_values('timestamp')
    
    # Get unique timestamps
    timestamps = df['timestamp'].unique()
    n = len(timestamps)
    
    # Split indices
    train_idx = int(n * train_ratio)
    buffer_idx = int(n * (train_ratio + buffer_ratio))
    
    # Get timestamps for each split
    train_timestamps = timestamps[:train_idx]
    buffer_timestamps = timestamps[train_idx:buffer_idx]
    test_timestamps = timestamps[buffer_idx:]
    
    # Create masks
    train_mask = df['timestamp'].isin(train_timestamps)
    test_mask = df['timestamp'].isin(test_timestamps)
    
    # Create splits
    train_df = df[train_mask]
    test_df = df[test_mask]
    
    print(f"Train set: {len(train_df)} samples from {train_timestamps.min()} to {train_timestamps.max()}")
    print(f"Test set: {len(test_df)} samples from {test_timestamps.min()} to {test_timestamps.max()}")
    
    return train_df, test_df

def prepare_xy(df, binary=False):
    # Remove non-feature columns
    non_features = ['timestamp', 'machine_id', 'failure_warning']
    X = df.drop(columns=non_features, errors='ignore')
    
    if binary:
        # Convert to binary classification (any warning vs no warning)
        y = (df['failure_warning'] > 0).astype(int)
    else:
        # Multi-class classification (0: no warning, 1: early, 2: imminent)
        y = df['failure_warning']
    
    return X, y

# Feature selection
def select_features(X_train, y_train, X_test, threshold=0.01):
    gbm = GradientBoostingClassifier(random_state=42)
    gbm.fit(X_train, y_train)
    
    # Feature importance
    feature_selector = SelectFromModel(gbm, threshold=threshold, prefit=True)
    X_train_selected = feature_selector.transform(X_train)
    X_test_selected = feature_selector.transform(X_test)
    
    # Get selected feature names
    selected_indices = feature_selector.get_support(indices=True)
    feature_names = X_train.columns[selected_indices].tolist()
    
    print(f"Selected {len(feature_names)} out of {X_train.shape[1]} features")
    
    return X_train_selected, X_test_selected, feature_names

# Train and evaluate multiple models
def train_evaluate_models(X_train, y_train, X_test, y_test):
    models = {
        'Random Forest': RandomForestClassifier(n_estimators=200, random_state=42, class_weight='balanced'),
        'Gradient Boosting': GradientBoostingClassifier(n_estimators=200, random_state=42)
    }
    
    results = {}
    
    for name, model in models.items():
        print(f"\nTraining {name}...")
        model.fit(X_train, y_train)
        
        # Predict on test set
        y_pred = model.predict(X_test)
        
        # Calculate metrics
        print(f"\nClassification Report for {name}:")
        report = classification_report(y_test, y_pred, output_dict=True)
        print(classification_report(y_test, y_pred))
        
        print("Confusion Matrix:")
        conf_matrix = confusion_matrix(y_test, y_pred)
        print(conf_matrix)
        
        # ROC curve
        y_prob = model.predict_proba(X_test)[:, 1]
        fpr, tpr, _ = roc_curve(y_test, y_prob)
        roc_auc = auc(fpr, tpr)
        
        # Precision-Recall curve
        precision, recall, _ = precision_recall_curve(y_test, y_prob)
        pr_auc = auc(recall, precision)
        
        results[name] = {
            'model': model,
            'report': report,
            'conf_matrix': conf_matrix,
            'roc': {'fpr': fpr, 'tpr': tpr, 'auc': roc_auc},
            'pr': {'precision': precision, 'recall': recall, 'auc': pr_auc}
        }
    
    return results

# Function to analyze important features
def analyze_feature_importance(model, feature_names, top_n=20):
    if hasattr(model, 'feature_importances_'):
        importances = model.feature_importances_
        indices = np.argsort(importances)[::-1]
        
        plt.figure(figsize=(12, 8))
        plt.title(f'Top {top_n} Feature Importances')
        plt.bar(range(min(top_n, len(feature_names))), 
                importances[indices][:top_n], 
                align='center')
        plt.xticks(range(min(top_n, len(feature_names))), 
                  [feature_names[i] for i in indices[:top_n]], 
                  rotation=90)
        plt.tight_layout()
        plt.savefig('feature_importance.png')
        
        # Return the top features
        return [(feature_names[i], importances[i]) for i in indices[:top_n]]
    else:
        return []

# Function to make predictions on new data
def predict_failures(model, scaler, feature_selector, new_data):
    # Preprocess new data
    X_new = new_data.drop(columns=['machine_id', 'timestamp'], errors='ignore')
    X_new_scaled = scaler.transform(X_new)
    X_new_selected = feature_selector.transform(X_new_scaled)
    
    # Make predictions
    y_prob = model.predict_proba(X_new_selected)[:, 1]
    
    # Create prediction DataFrame
    prediction_df = pd.DataFrame({
        'machine_id': new_data['machine_id'],
        'timestamp': new_data['timestamp'],
        'failure_probability': y_prob
    })

    return prediction_df
